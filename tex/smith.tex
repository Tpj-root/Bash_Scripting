\documentclass{article}

\usepackage{tikz}

    \begin{tikzpicture}
        % Draw a smooth sine curve
        \begin{axis}[
            axis lines = middle,
            grid = major,
            xlabel = $x$,
            ylabel = $y$,
            domain=-2:2,
            samples=100,
            width=12cm, height=8cm,
            smooth
        ]
            \addplot[blue, thick] {sin(deg(2*pi*x))};
        \end{axis}
    \end{tikzpicture}
